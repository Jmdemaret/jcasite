// === Configuration ===================================================
// Adresse qui recevra les emails. Par défaut c'est le compte qui déploie
// le script. Ajoutez d'autres emails en les séparant par des virgules.
const RECIPIENT = 'secretariat.judoclubanderlecht@gmail.com';

// Sujet par défaut si le formulaire n'en fournit pas
const DEFAULT_SUBJECT = '[Site] Nouveau message';

// Domaine autorisé (origine). Laissez vide pour autoriser tout le monde.
// Exemple : 'https://judoclubanderlecht.be'
const ALLOWED_ORIGIN = '';
// =====================================================================


function doPost(e) {
  try {
    const params = e && e.parameter ? e.parameter : {};
    const subject = params._subject || DEFAULT_SUBJECT;
    const source  = params._source  || 'site';

    // Corps de l'email : toutes les paires clé/valeur lisiblement
    const lines = Object.keys(params)
      .filter(k => !k.startsWith('_'))
      .map(k => '  ' + k + ' : ' + params[k]);

    const body =
      'Nouveau message depuis ' + source + '\n' +
      'Date : ' + new Date().toLocaleString('fr-BE') + '\n\n' +
      '--- Contenu ---\n' +
      lines.join('\n') + '\n\n' +
      '--- Fin ---\n';

    const replyTo = params.email || params.Email || '';

    MailApp.sendEmail({
      to: RECIPIENT,
      subject: subject,
      body: body,
      name: 'Site Judo Club Anderlecht',
      replyTo: replyTo || RECIPIENT
    });

    return _json({ ok: true });
  } catch (err) {
    return _json({ ok: false, error: err.toString() });
  }
}

// Health check : ouvrir l'URL dans un navigateur doit afficher "OK"
function doGet() {
  return ContentService
    .createTextOutput('Judo Club Anderlecht email endpoint — OK')
    .setMimeType(ContentService.MimeType.TEXT);
}

function _json(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
